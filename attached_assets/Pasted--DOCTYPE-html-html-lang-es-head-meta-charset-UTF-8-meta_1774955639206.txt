<script>
(function () {
  var wrap  = document.getElementById('hsWrap');
  var track = document.getElementById('hsTrack');
  var items = Array.from(document.querySelectorAll('.hs-item'));

  var CLOSED_W = 72;
  var GAP      = 12;
  /* offset = number of cards currently closed as strips */
  var offset   = 1; /* start with card 0 closed */

  function isMobile() { return window.innerWidth <= 768; }

  /* ── Desktop apply ── */
  function applyDesktop(animate) {
    track.style.transition = animate
      ? 'transform 0.55s cubic-bezier(0.4,0,0.2,1)'
      : 'none';
    track.style.transform = 'translateX(0)';

    items.forEach(function(item, i) {
      item.classList.toggle('hs-closed', i < offset);
    });

    if (!animate) {
      requestAnimationFrame(function() {
        requestAnimationFrame(function() {
          track.style.transition = '';
        });
      });
    }
  }

  /* ── Mobile apply — accordion, all cards visible stacked ── */
  function applyMobile() {
    track.style.transform = 'none';
    track.style.transition = 'none';
    /* On mobile: only the FIRST card (index 0) starts closed.
       All others start open. Tapping a closed card opens it;
       tapping an open card closes it. Independent toggles. */
    items.forEach(function(item, i) {
      /* initial state: card 0 closed, rest open */
    });
  }

  function apply(animate) {
    if (isMobile()) { applyMobile(); return; }
    applyDesktop(animate);
  }

  /* ── Desktop nav buttons ── */
  document.getElementById('hsNext').addEventListener('click', function() {
    if (isMobile()) return;
    if (offset < items.length - 1) { offset++; applyDesktop(true); }
  });
  document.getElementById('hsPrev').addEventListener('click', function() {
    if (isMobile()) return;
    if (offset > 0) { offset--; applyDesktop(true); }
  });

  /* ── Desktop: click open card → collapse it ── */
  /* ── Desktop: click closed strip → re-open back to that point ── */
  items.forEach(function(item) {
    item.addEventListener('click', function() {
      if (isMobile()) {
        /* Mobile accordion: toggle this card open/closed independently */
        this.classList.toggle('hs-closed');
        return;
      }
      var idx = parseInt(this.getAttribute('data-index'));
      if (this.classList.contains('hs-closed')) {
        /* clicking a strip → re-open from here */
        offset = idx;
        applyDesktop(true);
      } else {
        /* clicking open card → collapse it */
        offset = idx + 1;
        applyDesktop(true);
      }
    });
  });

  var resizeTimer;
  window.addEventListener('resize', function() {
    clearTimeout(resizeTimer);
    resizeTimer = setTimeout(function() { apply(false); }, 120);
  });

  apply(false);
})();
</script>
